144 private void init(int k, int w, int dim) {
145 if (filename == null)
146 if ((dim < 1) || (dim > MAXDIM))
147 throw new IllegalArgumentException("Dimension for SobolSequence must be > 0 and <= " + MAXDIM);
148 else if (dim < 1)
149 throw new IllegalArgumentException("Dimension for SobolSequence must be > 0");
150 if (w > MAXBITS || k >= MAXBITS)
151 throw new IllegalArgumentException("One must have k < 31 and k <= w <= 31 for SobolSequence");
152 numCols = k;
153 numRows = k; // Used in DigitalNetBase2, to read and print matrices and for interlacing.
154 outDigits = w;
155 numPoints = (1 << k);
156 this.dim = dim;
157 normFactor = 1.0 / ((double) (1L << (outDigits)));
158 genMat = new int[dim * numCols]; // Each column is a w-bit integer.
159 initGenMat();
160 }

170 public SobolSequence(int n, int dim) {
171 numCols = MAXBITS; // Defined in PointSet.
172 while (((n >> numCols) & 1) == 0)
173 numCols--;
174 if (1 << numCols != n)
175 numCols++;
176 init(numCols, MAXBITS, dim);
177 }

271 public SobolSequence(String filename, int k, int w, int dim) {
272
273 poly_from_file = new int[dim];
274 for (int i = 0; i < dim; i++)
275 poly_from_file[i] = 0;
276
277 minit_from_file = new int[dim][MAXDEGREE];
278 for (int i = 0; i < dim; i++) {
279 for (int j = 0; j < MAXDEGREE; j++) {
280 minit_from_file[i][j] = 0;
281 }
282 }
283
284 // Dimension d = 1
285 int d = 1;
286 poly_from_file[0] = 1;
287
288 // Read the direction number file up to a certain number of dimension dim
289 try {
290 // If filename can be found starting from the program's directory,
291 // it will be used; otherwise, the filename in the Jar archive will
292 // be used.
293 int prev_d = 1;
294
295 BufferedReader reader;
296
297 if (filename.startsWith("http") || filename.startsWith("ftp"))
298 reader = DigitalNetFromFile.openURL(filename);
299 else
300 reader = new BufferedReader(new FileReader(filename));
301
302 // First line of file is a comment; discard it
303 String line = reader.readLine();
304 String[] tokens;
305
306 while ((line = reader.readLine()) != null) {
307 tokens = line.split("[\t ]+");
308
309 // Direction number lines from dimension 2 and up
310 if (tokens.length < 4) {
311 System.err.println("\nBad direction number file format, < 4 tokens\n");
312 System.exit(1);
313 }
314
315 // Parse dim d, polynomial degree s and coefficients a
316 d = Integer.parseInt(tokens[0]);
317
318 int s = Integer.parseInt(tokens[1]);
319 int a = Integer.parseInt(tokens[2]);
320
321 if (s + 3 != tokens.length) {
322 System.err.println("\nBad direction number file format, not s+3 tokens!\n");
323 System.exit(1);
324 }
325
326 if (d != prev_d + 1) {
327 System.err.println("Dimension in file shall be in ");
328 System.err.println("increasing order, one per line!");
329 System.exit(1);
330 }
331 prev_d = d;
332
333 // If d in the file exceeds dim, stop reading!
334 if (d > dim)
335 break;
336
337 // We add a 1 on each side! Those '1' had been removed in the file.
338 poly_from_file[d - 1] = (1 << s) ^ (a << 1) ^ 1;
339
340 // Parse the s direction numbers
341 for (int i = 0; i < s; i++)
342 minit_from_file[d - 2][i] = Integer.parseInt(tokens[i + 3]);
343 } // end while
344
345 } catch (MalformedURLException e) {
346 System.err.println(" Invalid URL address: " + filename);
347 System.exit(1);
348
349 } catch (IOException e) {
350 System.err.println("Error: " + e);
351 System.exit(1);
352 }
353
354 if (dim > d) {
355 System.err.printf("\n\nNot enough dimension in file: %s", filename);
356 System.exit(1);
357 }
358 this.filename = filename;
359 init(k, w, dim);
360 }

368 public void extendSequence(int k) {
369 int start, degree, nextCol;
370 int i, j, c;
371 int oldNumCols = numCols;
372 int[] oldGenMat = genMat; // Save old generating matrix.
373
374 numCols = k;
375 numPoints = (1 << k);
376 genMat = new int[dim * numCols];
377
378 // the first dimension, j = 0.
379 for (c = 0; c < numCols; c++)
380 genMat[c] = (1 << (outDigits - c - 1));
381
382 // the other dimensions j > 0.
383 for (j = 1; j < dim; j++) {
384 // find the degree of primitive polynomial f_j
385 for (degree = MAXDEGREE; ((poly[j] >> degree) & 1) == 0; degree--)
386 ;
387 // Get initial direction numbers m_{j,0},..., m_{j,degree-1}.
388 start = j * numCols;
389 for (c = 0; (c < degree && c < numCols); c++)
390 genMat[start + c] = minit[j - 1][c] << (outDigits - c - 1);
391
392 // Compute the following ones via the recursion.
393 for (c = degree; c < numCols; c++) {
394 if (c < oldNumCols)
395 genMat[start + c] = oldGenMat[j * oldNumCols + c];
396 else {
397 nextCol = genMat[start + c - degree] >> degree;
398 for (i = 0; i < degree; i++)
399 if (((poly[j] >> i) & 1) == 1)
400 nextCol ^= genMat[start + c - degree + i];
401 genMat[start + c] = nextCol;
402 }
403 }
404 }
405 }

407 // Initializes the original generator matrices for a sequence.
408 private void initGenMat() {
409 int start, degree, nextCol;
410 int i, j, c;
411
412 // the first dimension, j = 0. This is the identity matrix.
413 for (c = 0; c < numCols; c++)
414 genMat[c] = (1 << (outDigits - c - 1));
415
416 // the other dimensions j > 0.
417 for (j = 1; j < dim; j++) {
418 // if a direction number file was provided, use it
419 int polynomial = (filename != null ? poly_from_file[j] : poly[j]);
420 // find the degree of primitive polynomial f_j
421 for (degree = MAXDEGREE; ((polynomial >> degree) & 1) == 0; degree--)
422 ;
423 // Get initial direction numbers m_{j,0},..., m_{j,degree-1}.
424 start = j * numCols;
425 for (c = 0; (c < degree && c < numCols); c++) {
426 int m_i = (filename != null ? minit_from_file[j - 1][c] : minit[j - 1][c]);
427 genMat[start + c] = m_i << (outDigits - c - 1);
428 }
429
430 // Compute the following ones via the recursion.
431 for (c = degree; c < numCols; c++) {
432 nextCol = genMat[start + c - degree] >> degree;
433 for (i = 0; i < degree; i++)
434 if (((polynomial >> i) & 1) == 1)
435 nextCol ^= genMat[start + c - degree + i];
436 genMat[start + c] = nextCol;
437 }
438 }
439 }
